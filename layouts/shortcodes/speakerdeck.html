{{- $url := .Get "url" -}} {{- $id := .Get "id" -}} {{- if not (or $url $id) -}} {{- errorf "speakerdeck shortcode requires either 'url' or 'id' parameter" -}} {{- end -}} {{- $slideId := "" -}} {{- if $id -}} {{- $slideId = $id -}} {{- else if $url -}} {{- $parts := split $url "/" -}} {{- if ge (len $parts) 2 -}} {{- $slideId = index $parts (sub (len $parts) 1) -}} {{- end -}} {{- end -}}
