{{- $url := .Get "url" -}} {{- $id := .Get "id" -}} {{- if not (or $url $id) -}} {{- errorf "docswell shortcode requires either 'url' or 'id' parameter" -}} {{- end -}} {{- $slideUrl := "" -}} {{- if $url -}} {{- $slideUrl = $url -}} {{- else if $id -}} {{- $slideUrl = printf "https://www.docswell.com/s/%s" $id -}} {{- end -}}
